# 相逢 (Xiangfeng) 数据库架构方案

## 一、数据库架构概述

### 1.1 技术选型
- **数据库平台**: Supabase (基于 PostgreSQL 的全托管后端即服务)
- **核心功能**: 实时数据库、身份认证、文件存储、边缘函数
- **ORM框架**: Prisma (类型安全、自动生成客户端)
- **迁移工具**: Supabase CLI + Prisma Migrate (数据库版本控制)
- **连接管理**: Supabase 自动连接池管理
- **备份策略**: Supabase 自动备份 (7天到30天保留期)
- **实时功能**: PostgreSQL 逻辑复制 + Supabase Realtime
- **边缘缓存**: Supabase 全球 CDN + Redis 缓存层

```prisma
model User {
  id                String    @id @default(cuid())
  email             String    @unique @db.VarChar(255)

  website           String?   @db.VarChar(255)
  location          String?   @db.VarChar(100)
  
  // 认证相关
  password          String    @db.VarChar(255)
  emailVerified     Boolean   @default(false)
  verificationToken String?   @db.VarChar(255)
  resetToken        String?   @db.VarChar(255)
  resetTokenExpiry  DateTime?
  
  // 状态管理
  status            UserStatus @default(ACTIVE)
  role              UserRole   @default(USER)
  onlineStatus      OnlineStatus @default(OFFLINE)

  likes             Like[]

  following         Follow[]  @relation("UserFollowing")
  notifications     Notification[]
  settings          UserSettings?
  
  @@index([email])
  @@index([username])
  @@index([status])
  @@index([createdAt])
}

// 用户关注关系表
model Follow {
  id          String   @id @default(cuid())

  @@index([followingId])

  allowMention      Boolean    @default(true)
```

```prisma
// 文章主表
model Article {
  id          String        @id @default(cuid())
  title       String        @db.VarChar(255)
  slug        String        @unique @db.VarChar(255)
  excerpt     String        @db.VarChar(500)
  content     String        @db.Text // Markdown/HTML内容
  coverImage  String?       @db.Text
  
  // 作者信息
  authorId    String
  author      User          @relation(fields: [authorId], references: [id], onDelete: Cascade)
  
  // 文章状态
  status      ArticleStatus @default(DRAFT)
  visibility  Visibility    @default(PUBLIC)
  
  // 内容分类
  category    String?       @db.VarChar(50)
  tags        Tag[]         @relation("ArticleTags")
  
  // 互动统计
  likesCount     Int       @default(0)
  commentsCount  Int       @default(0)
  viewsCount     Int       @default(0)
  sharesCount    Int       @default(0)
  bookmarksCount Int       @default(0)
  
  // 内容质量
  wordCount   Int           @default(0)
  readingTime Int          @default(0) // 预计阅读时间（分钟）
  
  // 发布时间控制
  publishedAt DateTime?
  scheduledAt DateTime? // 定时发布
  
  // 时间戳
  createdAt   DateTime      @default(now())
  updatedAt   DateTime      @updatedAt
  deletedAt   DateTime?     // 软删除
  
  // 关系定义
  comments    Comment[]
  likes       Like[]
  bookmarks   Bookmark[]
  views       View[]
  
  @@index([authorId])
  @@index([status])
  @@index([publishedAt])
  @@index([createdAt])
  @@index([category])
}

// 标签表
model Tag {
  id          String   @id @default(cuid())
  name        String   @unique @db.VarChar(50)
  slug        String   @unique @db.VarChar(50)
  description String?  @db.VarChar(255)
  color       String?  @db.VarChar(7) // 十六进制颜色代码
  
  // 统计信息
  articlesCount Int @default(0)
  
  // 时间戳
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
  
  // 关系定义
  articles Article[] @relation("ArticleTags")
  
  @@index([name])
  @@index([slug])
}

// 评论表
model Comment {
  id        String   @id @default(cuid())
  content   String   @db.Text
  
  // 关联信息
  articleId String
  authorId  String
  parentId  String?  // 回复的评论ID
  
  // 评论状态
  status    CommentStatus @default(PUBLISHED)
  
  // 时间戳
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  deletedAt DateTime? // 软删除
  
  // 关系定义
  article Article @relation(fields: [articleId], references: [id], onDelete: Cascade)
  author  User    @relation(fields: [authorId], references: [id], onDelete: Cascade)
  parent  Comment? @relation("CommentReplies", fields: [parentId], references: [id])
  replies Comment[] @relation("CommentReplies")
  likes   Like[]
  
  @@index([articleId])
  @@index([authorId])
  @@index([parentId])
  @@index([status])
  @@index([createdAt])
}

// 点赞表（支持文章和评论）
model Like {
  id        String     @id @default(cuid())
  userId    String
  targetId  String     // 文章ID或评论ID
  targetType LikeType  // 点赞目标类型
  createdAt DateTime   @default(now())
  
  user User @relation(fields: [userId], references: [id], onDelete: Cascade)
  
  @@unique([userId, targetId, targetType])
  @@index([userId])
  @@index([targetId])
  @@index([targetType])
}

// 收藏表
model Bookmark {
  id        String   @id @default(cuid())
  userId    String
  articleId String
  createdAt DateTime @default(now())
  
  user    User    @relation(fields: [userId], references: [id], onDelete: Cascade)
  article Article @relation(fields: [articleId], references: [id], onDelete: Cascade)
  
  @@unique([userId, articleId])
  @@index([userId])
  @@index([articleId])
  @@index([createdAt])
}

// 浏览记录表
model View {
  id        String   @id @default(cuid())
  articleId String
  userId    String?  // 未登录用户可为null
  ipAddress String?  @db.VarChar(45)
  userAgent String?  @db.Text
  referrer  String?  @db.Text
  createdAt DateTime @default(now())
  
  article Article @relation(fields: [articleId], references: [id], onDelete: Cascade)
  user    User?   @relation(fields: [userId], references: [id], onDelete: SetNull)
  
  @@index([articleId])
  @@index([userId])
  @@index([createdAt])
}
```

### 6.3 边缘函数实现
```typescript
// Supabase 边缘函数 - 文章统计更新
import { serve } from 'https://deno.land/std@0.168.0/http/server.ts'
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'

serve(async (req) => {
  try {
    const { articleId } = await req.json()
    
    // 创建 Supabase 客户端
    const supabaseClient = createClient(
      Deno.env.get('SUPABASE_URL') ?? '',
      Deno.env.get('SUPABASE_SERVICE_KEY') ?? ''
    )

    // 更新文章统计
    const { data, error } = await supabaseClient
      .from('Article')
      .update({ 
        viewsCount: supabaseClient.sql`views_count + 1`,
        lastViewedAt: new Date().toISOString()
      })
      .eq('id', articleId)
      .select()
      .single()

    if (error) throw error

    return new Response(
      JSON.stringify({ success: true, data }),
      { headers: { 'Content-Type': 'application/json' } }
    )
  } catch (error) {
    return new Response(
      JSON.stringify({ error: error.message }),
      { status: 500, headers: { 'Content-Type': 'application/json' } }
    )
  }
})
```

### 6.4 文件存储集成
```typescript
// Supabase 文件存储 - 头像上传
import { createClient } from '@supabase/supabase-js'

const supabase = createClient(process.env.NEXT_PUBLIC_SUPABASE_URL, process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY)

// 上传用户头像
const uploadAvatar = async (file: File, userId: string) => {
  try {
    const fileName = `avatars/${userId}/${Date.now()}-${file.name}`
    
    // 上传文件到 Supabase Storage
    const { data, error } = await supabase.storage
      .from('avatars')
      .upload(fileName, file, {
        cacheControl: '3600',
        upsert: false
      })

    if (error) throw error

    // 获取公开访问URL
    const { data: { publicUrl } } = supabase.storage
      .from('avatars')
      .getPublicUrl(fileName)

    // 更新用户头像URL
    const { error: updateError } = await supabase
      .from('User')
      .update({ avatar: publicUrl })
      .eq('id', userId)

    if (updateError) throw updateError

    return { success: true, url: publicUrl }
  } catch (error) {
    return { success: false, error: error.message }
  }
}

// 生成图片缩略图
const generateThumbnail = async (file: File) => {
  // 使用 Supabase 图片转换功能
  const { data, error } = await supabase.storage
    .from('images')
    .upload(`thumbnails/${file.name}`, file, {
      transform: {
        width: 300,
        height: 300,
        resize: 'cover'
      }
    })

  return { data, error }
}
```

## 七、总结

本数据库架构方案基于 Supabase 平台为相逢项目提供了现代化的数据存储解决方案，具有以下核心优势：

### 7.1 Supabase 平台优势
1. **Serverless 架构**: 无需管理基础设施，自动扩缩容，按需付费
2. **实时能力**: 内置 WebSocket 支持，实现数据的实时同步和推送
3. **全栈集成**: 数据库、认证、存储、边缘函数一体化解决方案
4. **全球部署**: 多区域部署，就近访问，降低延迟
5. **企业级安全**: SSL 加密、自动备份、合规认证

### 7.2 技术特性
1. **完整性**: 涵盖用户系统、内容管理、社区互动、系统管理等全业务模块
2. **扩展性**: Serverless 架构天然支持弹性扩展，无需容量规划
3. **安全性**: RLS 行级安全、JWT 认证、自动备份和恢复
4. **性能**: 连接池自动管理、全球 CDN、边缘计算加速
5. **可维护性**: 可视化 Dashboard、自动化运维、完整监控告警

### 7.3 开发效率提升
- **快速开发**: 内置认证、存储、实时功能，减少重复开发
- **类型安全**: Prisma + TypeScript 提供完整的类型支持
- **实时同步**: 前后端数据自动同步，简化状态管理
- **边缘计算**: 就近处理业务逻辑，提升响应速度
- **自动化工具**: CLI 工具、迁移管理、性能监控

### 7.4 成本优化
- **按需付费**: 无服务器架构，只为实际使用付费
- **免运维**: 无需 DBA 和运维团队，降低人力成本
- **自动优化**: 平台自动进行性能调优和容量管理
- **开发加速**: 减少基础设施开发时间，专注业务逻辑

通过采用 Supabase 平台，相逢项目能够快速构建高性能、可扩展、安全可靠的应用，为从初创到大规模用户的全生命周期发展提供强有力的技术支撑。
